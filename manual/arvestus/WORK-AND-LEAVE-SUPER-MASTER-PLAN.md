```sql
  created_by UUID REFERENCES employees(id)

  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  tenant_id UUID REFERENCES tenants(id),
  
  employee_id UUID NOT NULL REFERENCES employees(id),
  leave_type_id UUID NOT NULL REFERENCES leave_types(id),
  year INTEGER NOT NULL,
  
  -- Balance tracking
  total_days DECIMAL(5,2) NOT NULL DEFAULT 0, -- Total entitlement
  used_days DECIMAL(5,2) NOT NULL DEFAULT 0, -- Already used

  UNIQUE(employee_id, leave_type_id, year)
```

```sql
-- Calculate working days between dates (excluding weekends and holidays)
CREATE OR REPLACE FUNCTION calculate_leave_working_days(
  p_start_date DATE,
  p_end_date DATE,
  p_country_code TEXT DEFAULT 'EE'
) RETURNS INTEGER AS $$
DECLARE
  working_days INTEGER := 0;
  current_date DATE := p_start_date;
BEGIN
  WHILE current_date <= p_end_date LOOP
    -- Skip weekends
    IF EXTRACT(DOW FROM current_date) NOT IN (0, 6) THEN
      -- Skip holidays
      IF NOT is_holiday(current_date, p_country_code) THEN
        working_days := working_days + 1;
      END IF;
    END IF;
    
    current_date := current_date + INTERVAL '1 day';
  END LOOP;
  
  RETURN working_days;
END;
$$ LANGUAGE plpgsql STABLE;

-- Update leave balance
CREATE OR REPLACE FUNCTION update_leave_balance()
RETURNS TRIGGER AS $$
DECLARE
  balance_record RECORD;
BEGIN
  -- Get or create balance record
  SELECT * INTO balance_record
  FROM leave_balances
  WHERE employee_id = NEW.employee_id
    AND leave_type_id = NEW.leave_type_id
    AND year = EXTRACT(YEAR FROM NEW.start_date);
  
  -- Update based on status change
  IF NEW.status = 'approved' AND OLD.status = 'pending' THEN
    -- Move from pending to planned
    IF NEW.start_date > CURRENT_DATE THEN
      UPDATE leave_balances
      SET planned_days = planned_days + NEW.working_days
      WHERE id = balance_record.id;
    ELSE
      -- Already started or past, mark as used
      UPDATE leave_balances
      SET used_days = used_days + NEW.working_days
      WHERE id = balance_record.id;
    END IF;
  ELSIF NEW.status = 'rejected' OR NEW.status = 'cancelled' THEN
    -- Return days
    IF OLD.status = 'approved' THEN
      IF NEW.start_date > CURRENT_DATE THEN
        UPDATE leave_balances
        SET planned_days = planned_days - NEW.working_days
        WHERE id = balance_record.id;
      ELSE
        UPDATE leave_balances
        SET used_days = used_days - NEW.working_days
        WHERE id = balance_record.id;
      END IF;
    END IF;
  END IF;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger
CREATE TRIGGER leave_request_balance_update
  AFTER UPDATE ON leave_requests
  FOR EACH ROW
  WHEN (OLD.status IS DISTINCT FROM NEW.status)
  EXECUTE FUNCTION update_leave_balance();

RETURNS TRIGGER AS $$
BEGIN
  -- Notify employee

      ELSE 'leave_updated'
    END,
    CASE NEW.status
      WHEN 'approved' THEN 'Puhkusetaotlus kinnitatud'
      WHEN 'rejected' THEN 'Puhkusetaotlus tagasi lÃ¼katud'
      ELSE 'Puhkusetaotlus muudetud'
    END,
    CASE NEW.status
      WHEN 'approved' THEN format('Sinu puhkusetaotlus %s - %s on kinnitatud', NEW.start_date, NEW.end_date)
      WHEN 'rejected' THEN format('Sinu puhkusetaotlus %s - %s on tagasi lÃ¼katud. PÃµhjus: %s', NEW.start_date, NEW.end_date, NEW.rejection_reason)
      ELSE format('Sinu puhkusetaotlus %s - %s on muudetud', NEW.start_date, NEW.end_date)
```
